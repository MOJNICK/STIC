	\frametitle{Cold, Confined Atomic Ions}
		\framesubtitle{Overview}

		{\normalsize
		\hspace{0.5cm}{Cold, Confined Atomic Ions technique uses laser beam to cool atomic ions, which are inside ion trap. After this process ions are organized in stable array. Qubit is represented with different levels of electron energy. We could pump ion with specific energy laser beam to change it state. Reading could be done in the same way.}\\
		}
		

\end{frame}

\begin{frame}
	\frametitle{Quantum Dot}
		\framesubtitle{Overview}

		{\normalsize
		\hspace{0.5cm}{Quantum Dots are a nano-scale crystals $(1^{-9}m)$. As the size of QD is similar to De Broglie wave length, the structure is a kind of a trap for specific energy, free electrons. There is also possible QD design, such that it could trap electrons by spin. } \\
		}
		

\end{frame}
